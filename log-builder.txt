--Convert selection into list of pathstell application "Music"	activate	set selectionList to selection	set n to 0	set songPath to null	set pathList to {}	repeat while n < length of selectionList		set n to n + 1		set songID to item n of selectionList		set songPath to get location of songID		set pathList to pathList & songPath	end repeat	set n to 0end tell--Get show invisibles)set bedPath to "/Users/Ben/Documents/Radio/Beds/Active/a"set beds to (list folder bedPath without invisibles)set talkMarker to
